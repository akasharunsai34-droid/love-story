<script>
  // pages logic (slide left/right)
  const pages = Array.from(document.querySelectorAll('.page'));
  let idx = 0;
  function show(i){
    pages.forEach((p,pi)=>{
      p.classList.remove('active','prev');
      if(pi === i) p.classList.add('active');
      else if(pi < i) p.classList.add('prev');
    });
    idx = i;
  }
  function goNext(){ show(Math.min(pages.length-1, idx+1)) }
  function goPrev(){ show(Math.max(0, idx-1)) }
  function restart(){ show(0) }

  // helper for buttons inside some cards
  function nextFrom(el){ goNext(); }

  // heart reveal (page 5)
  function revealLove(){
    const msg = document.getElementById('loveMsg');
    msg.style.display = 'block';
    msg.animate([{opacity:0, transform:'scale(.8)'},{opacity:1, transform:'scale(1)'}], {duration:420, fill:'forwards'});
    const h = document.getElementById('heart');
    h.animate([{transform:'scale(1)'},{transform:'scale(1.18)'},{transform:'scale(1)'}],{duration:420,fill:'forwards'});
  }

  // create QR for the whole page (link back to this file's URL so scanning opens full experience)
  new QRCode(document.getElementById('qrcode'), {
    text: window.location.href,
    width: 108,
    height: 108,
    colorDark : "#222",
    colorLight : "#ffffff",
    correctLevel : QRCode.CorrectLevel.H
  });

  // convenience: make the "Next ‚Üí" inside page1's pill work
  window.goNext = goNext;
  window.goPrev = goPrev;
  window.restart = restart;
  window.revealLove = revealLove;
</script>
